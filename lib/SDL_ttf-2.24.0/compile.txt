编译示例程序需要静态链接SDL2.lib和SDLmain.lib
cmake ../source -G "Visual Studio 17 2022" -DCMAKE_INSTALL_PREFIX=../install -DSDL2_DIR="F:\Code\#Files\Libraries\SDL-2.32.10-MSVC\install" -DSDL2_LIBRARY="F:\Code\#Files\Libraries\SDL-2.32.10-MSVC\install\lib\SDL2.lib" -DSDL2_INCLUDE_DIR="F:\Code\#Files\Libraries\SDL-2.32.10-MSVC\install\include\SDL2" -DSDL2_MAIN_LIBRARY="F:\Code\#Files\Libraries\SDL-2.32.10-MSVC\install\lib\SDL2main.lib"

cmake --build . --parallel --config Release

cmake --install . --config Release
